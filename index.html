var s2 = swoosh(document.getElementById("ex2_div"))
  .on('collideLeft', function(e){
    ex2_log.innerHTML += "collideLeft Event triggered<br>"
  })
  .on('collideTop', function(e){
    ex2_log.innerHTML += "collideTop Event triggered<br>"

    ex2_log.innerHTML += "collideRight Event triggered<br>"
  })

    ex2_log.innerHTML += "collideBottom Event triggered<br>"
  });
